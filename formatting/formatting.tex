% Logic & Custom Commands

\ExplSyntaxOn

\cs_new_protected:Npn \generate_label_from_name #1 {
	\tl_set:Nn \l_tmpa_tl {#1}
	\tl_set:Nx \l_tmpa_tl { \tl_lower_case:n { \l_tmpa_tl } }
	\tl_replace_all:Nnn \l_tmpa_tl { ~ } { _ }
	\label{chapter_\l_tmpa_tl}
}

\newcommand{\addchapter}[2]{
	\chapter{#2}
	\generate_label_from_name{#1}
	\input{content/#1}
}

\newcommand{\addcontent}[2]{
	\tl_if_in:nnT {#2} { b } { \newpage }
	\input{content/#1}
	\tl_if_in:nnT {#2} { a } { \newpage }
}
